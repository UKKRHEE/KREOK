country_event = {
	id = fprf.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event fprf.1"}
	title = prf.160.t
	desc = prf.160.d
	picture = GFX_report_event_PRF_war

	trigger = {
		OR = {
			tag = PRF
		}
		OR = {
			is_in_faction_with = KOR
		}
		has_war_with = DEH
		controls_state = 439 #Delhi
		controls_state = 443 #Sindh
		surrender_progress < 0.2
		DEH = {
			is_in_faction = yes
			NOT = { is_faction_leader = yes }
			NOT = { has_idea = DEH_considering_peace }
			surrender_progress > 0.8
		}
	}

	mean_time_to_happen = {
		days = 7
	}

	option = {
		name = prf.160.a #Propose an armistice!
		ai_chance = { factor = 100 }
		DEH = {
			add_timed_idea = { idea = DEH_considering_peace days = 180 }
		}
		random_country = {
			limit = {
				DEH = { is_in_faction_with = PREV }
				is_faction_leader = yes
			}
			country_event = { id = fprf.2 }
		}
	}
	option = {
		name = prf.160.b #We shall fight on!
		ai_chance = { factor = 0 }
		add_war_support = -0.30
	}
}
